////////////////////////////////////////////////////
//// SEND CAMERA IP TO SUPABASE
////////////////////////////////////////////////////

void updateCameraIP(){

  String ip = WiFi.localIP().toString();

  HTTPClient http;

  http.begin(supabaseURL);

  http.addHeader("Content-Type","application/json");
  http.addHeader("apikey", supabaseKey);
  http.addHeader("Authorization", String("Bearer ") + supabaseKey);

  String body = "{\"ip\":\""+ip+"\"}";

  http.sendRequest("PATCH", body);

  http.end();

  Serial.println("Camera IP sent to Supabase");
}

////////////////////////////////////////////////////
//// GET WIFI FROM SENSOR ESP32
////////////////////////////////////////////////////

void getWiFiFromSensor(){

  Serial.println("Connecting to Sensor AP...");

  WiFi.begin(sensorAP, sensorAPPass);

  while(WiFi.status()!=WL_CONNECTED){
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nConnected to Sensor AP");

  HTTPClient http;

  http.begin("http://192.168.4.1/wifi");

  int httpCode = http.GET();

  if(httpCode==200){

    String payload = http.getString();

    int s1 = payload.indexOf("\"ssid\":\"")+8;
    int s2 = payload.indexOf("\"",s1);

    routerSSID = payload.substring(s1,s2);

    int p1 = payload.indexOf("\"pass\":\"")+8;
    int p2 = payload.indexOf("\"",p1);

    routerPASS = payload.substring(p1,p2);

    Serial.println("Router SSID: "+routerSSID);
  }

  http.end();

  WiFi.disconnect(false);
  delay(1000);
}

////////////////////////////////////////////////////
//// CONNECT TO ROUTER
////////////////////////////////////////////////////

void connectRouter(){

  Serial.println("Connecting to Router...");

  WiFi.begin(routerSSID.c_str(), routerPASS.c_str());

  while(WiFi.status()!=WL_CONNECTED){
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nRouter Connected!");

  Serial.print("Camera Stream URL: http://");
  Serial.print(WiFi.localIP());
  Serial.println("/stream");
}

////////////////////////////////////////////////////
//// STREAM HANDLER
////////////////////////////////////////////////////

static esp_err_t stream_handler(httpd_req_t *req){

  camera_fb_t * fb = NULL;
  esp_err_t res = ESP_OK;

  httpd_resp_set_type(req,"multipart/x-mixed-replace; boundary=frame");
  httpd_resp_set_hdr(req,"Access-Control-Allow-Origin","*");

  while(true){

    fb = esp_camera_fb_get();

    if(!fb){
      Serial.println("Camera capture failed");
      res = ESP_FAIL;
      break;
    }

    res = httpd_resp_send_chunk(req,"\r\n--frame\r\n",9);

    if(res != ESP_OK){
      esp_camera_fb_return(fb);
      break;
    }

    char header[64];

    int hlen = snprintf(header,64,
      "Content-Type: image/jpeg\r\nContent-Length: %u\r\n\r\n",
      fb->len);

    res = httpd_resp_send_chunk(req,header,hlen);

    if(res != ESP_OK){
      esp_camera_fb_return(fb);
      break;
    }

    res = httpd_resp_send_chunk(req,(const char*)fb->buf,fb->len);

    esp_camera_fb_return(fb);

    if(res != ESP_OK){
      break;
    }

    delay(3);
  }

  Serial.println("Stream client disconnected");

  return res;
}

////////////////////////////////////////////////////
//// START CAMERA SERVER
////////////////////////////////////////////////////

void startCameraServer(){

  httpd_config_t config = HTTPD_DEFAULT_CONFIG();

  config.server_port = 80;

  httpd_uri_t stream_uri = {
    .uri = "/stream",
    .method = HTTP_GET,
    .handler = stream_handler,
    .user_ctx = NULL
  };

  if(httpd_start(&stream_httpd,&config)==ESP_OK){
    httpd_register_uri_handler(stream_httpd,&stream_uri);
  }
}

////////////////////////////////////////////////////
//// SETUP
////////////////////////////////////////////////////

void setup(){

  Serial.begin(115200);

  camera_config_t config;

  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;

  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;

  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;

  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;

  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;

  config.xclk_freq_hz = 20000000;
  config.pixel_format = PIXFORMAT_JPEG;

  if(psramFound()){

    config.frame_size = FRAMESIZE_SVGA;   // 800x600
    config.jpeg_quality = 6;              // very sharp
    config.fb_count = 2;

  }else{

    config.frame_size = FRAMESIZE_VGA;
    config.jpeg_quality = 8;
    config.fb_count = 1;
  }

  if(esp_camera_init(&config)!=ESP_OK){
    Serial.println("Camera Init Failed");
    return;
  }

  sensor_t * s = esp_camera_sensor_get();

  if(s){

    s->set_vflip(s,1);
    s->set_hmirror(s,1);

    s->set_brightness(s,1);
    s->set_contrast(s,2);
    s->set_saturation(s,1);

    s->set_sharpness(s,2);
    s->set_denoise(s,1);

    s->set_gain_ctrl(s,1);
    s->set_exposure_ctrl(s,1);
    s->set_whitebal(s,1);
    s->set_awb_gain(s,1);
  }

  getWiFiFromSensor();

  connectRouter();

  updateCameraIP();

  startCameraServer();
}
